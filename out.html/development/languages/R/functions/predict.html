# X11()

d <- data.frame (
  t <- c(25, 22, 18, 19, 21, 26, 26, 27, 24, 22),
  w <- c( 4,  3,  0,  6,  5,  3,  4,  1,  3,  1)
)

At =  3
Bt =  4
Et = rnorm(length(t), mean = 0, sd = 3.2)

Aw =  5
Bw = -2
Ew = rnorm(length(t), mean = 0, sd = 0.9)

d$o <- At + Bt*t + Et +
       Aw + Bw*w + Ew


model <- lm(d$o ~ d$t * d$w)


prediction <- predict(model)


data.frame(
  p = prediction,
  o = d$o
)
