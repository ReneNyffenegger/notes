options("digits"=2)

m <- matrix (
       c(  1, 3, 1,
           7, 3, 5),
       ncol  = 3,
       byrow = TRUE
)

prop.table(m)
#      [,1] [,2] [,3]
# [1,] 0.05 0.15 0.05
# [2,] 0.35 0.15 0.25

prop.table(m, 1)
