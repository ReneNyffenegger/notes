use warnings;
use strict;
use feature 'say';

use String::Diff;

my ($one, $two) = String::Diff::diff (
                    'Well, a bit here and a bit there',
                    'ok, a bit in San Francisco and a bit in New York',
                    remove_open  => '',
                    remove_close => '',
                    append_open  => '',
                    append_close => '');

say $one;
say $two;
__END__
