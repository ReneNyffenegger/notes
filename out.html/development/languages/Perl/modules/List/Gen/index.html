use warnings;
use strict;
use feature 'say';


use List::Gen '*';

# by # {{{

for my $x (every 3 => (1, 2, 3, 4, 5, 6, 7, 8, 9)) {
  say join " - ", @$x;
}
# 1 - 2 - 3
# 4 - 5 - 6
# 7 - 8 - 9

# }}}


# mapn # {{{
print "\n";
print mapn { "$_ : " . (join ' - ', @_[ 1 .. $#_ ]) . "\n" } 4 => qw(foo 1 2 3 bar 4 5 6 baz 7 8 9);
# foo : 1 - 2 - 3
# bar : 4 - 5 - 6
# baz : 7 - 8 - 9

print "\n";
# }}}
