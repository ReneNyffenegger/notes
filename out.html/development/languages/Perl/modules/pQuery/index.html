use warnings;
use strict;

use pQuery;

my $html=<<"END"; # {{{


  
    
  
  

    
 aaa 


    
 
 bbb 
 


    
 ccc 


    

      
ab

      
cd

    


  


END
# }}}

my $p = pQuery($html);

print $p -> find('p'  ) -> html, "\n";
$p -> find('div')->each(sub {
  print 'Each: ' . pQuery($_)->html,"\n";
});

print $p -> find('div')->eq(2) -> text;

print "style: ", $p -> find('style')->text, "\n";

# Note: with ...->html, the enclosing ..
 elements wouldn't
#       be printed:
print "table: ", $p -> find('table')->toHtml;
