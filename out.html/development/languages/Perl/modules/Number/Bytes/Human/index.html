sub show { # {{{
    
    my $bytes = shift;

    printf("  %10i   %10s   %10s   %10s\n", 
      $bytes, 
      format_bytes($bytes, {'si' =>  1, base => 1000   }),
      format_bytes($bytes, {'si' =>  1, base => 1024   }),
      format_bytes($bytes, {'si' =>  0, base => 1024000}),
    );

} # }}}
