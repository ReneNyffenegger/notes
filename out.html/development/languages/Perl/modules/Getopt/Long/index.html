use strict;
use warnings;

use Getopt::Long;

use vars qw($opt);
$opt = { "help" => \&Usage, };

Getopt::Long::GetOptions(
    $opt,
    "help",
    "flag_1=s",
    "other_flag=s",
);

print("\n");

for my $option_given (qw{flag_1 other_flag}) {
  if (exists $opt->{$option_given}) {
    printf("Option %-10s: %s\n", $option_given, $opt->{$option_given} );
  }
}

print("\n");


sub Usage {
  print STDERR <<"USAGE";

Usage: perl $0 [options]

Possible options are:

  --help                 This help message

  --flag_1=      Use  for the flag you want 
                         to specify.

  --other_flag=   Use  as you may wish.

USAGE
}
