use warnings;
use strict;

use XML::XSLT;

my $xml = q{


  This is the Title  

  This is the heading

  28.08.2015         

  foo                
  bar                
  baz                


};

my $xsl = q{

  
    

      

        <xsl:value-of select="title"/>

      

      

        


        


        
    
        
          
    • 
            
        


      
    
  

  
    

    Date:  
    

  


};

my $xslt = XML::XSLT->new ($xsl, warnings => 1);

$xslt->transform ($xml);

open (my $html, '>', 'out.html') or die;
print $html $xslt->toString;
close $html;

$xslt->dispose();
