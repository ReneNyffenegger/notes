Convert ESRI Ascii to xyz format

#!/usr/bin/perl -anl

$g++ or $, = "\t", $. -= 7;

$. < 0 and ( $p{$F[0]} = $F[1], next );

print ( ($p{'xllcorner'} + $_ * $p{'cellsize'}), ($p{'yllcorner'} + ($p{'nrows'} - 1 - $.) * $p{'cellsize'} ), $F[$_]) for (0..$#F);
